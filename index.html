Object:

Classic Perl Object

package Duke;
sub new {
    my ( $class, %data ) = @_;
    return bless \%data, $class;
}

Instantiate an Object

use Duke;
my $duke = Duke->new(
    foo => "bar",
    baz => "fuzz",
);

say $duke->{foo}; # "bar"

Blessed

use Scalar::Util qw( blessed );

Package name is attached to the reference

Attributes

package Duke;
sub new {
    my ( $class, %args ) = @_;
    return bless {
        src   => $args{src},    # Extract
        xform => $args{xform},  # Transform
        dest  => $args{dest},   # Load
    }, $class;
}

Mutators

Methods to get/set attributes

package Duke;
sub src {
    my ( $self ) = @_;
    return $self->{src};
}
sub xform {
    my ( $self ) = @_;
    return $self->{xform};
}
sub dest {
    my ( $self, $new_dest ) = @_;
    if ( $new_dest ) {
        $self->{dest} = $new_dest;
    }
    return $self->{dest};
}

Allow new destinations, but not new sources or transforms.

Methods

package Duke;
sub add_xform {
    my ( $self, $new_xform ) = @_;
    push @{ $self->{xform} }, $new_xform;
    return;
}

Allow appending extra transform steps.
